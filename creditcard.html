from sklearn.linear_model import LogisticRegression
from sklearn.metrics import accuracy_score
from sklearn.metrics import confusion_matrix
from sklearn.metrics import classification_report

models=[]
models.append(('LR',LogisticRegression(solver='liblinear', random_state=101))) #LR 参数
#the for loop used here will help if multiple models are being run。
for name, model in models:
    model.fit(X_train,y_train) #train the data
    predictions = model.predict(X_test) #make prediction
    ac_score=accuracy_score(y_test,predictions) #compare the prediction & testing set result, get the prediction accuracy
    msg="%s: %f" % (name, ac_score) #generate the LR  report
    print(msg)  # the model name & prediction accuracy
    print(confusion_matrix(y_test,predictions)) #cm
    print(classification_report(y_test,predictions))#classification report
